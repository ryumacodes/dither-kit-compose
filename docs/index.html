01

Area + line

DesktopMobile
AreaChart(data, bloom = Bloom.Aura) {
    grid()
    xAxis()
    yAxis()
    legend(isClickable = true)
    tooltip()
    area("desktop", color = DitherColor.Blue)
    line("mobile", color = DitherColor.Purple)
}

LineChart(data, markerIndex = selected) {
    grid(vertical = true)

        "desktop",
        color = DitherColor.Pink,
        activeDot = DotStyle(DotVariant.Filled),
    )
    referenceLine(200, label = "Target")
